CREATE TABLE mfMink
(color varchar(20),
furLength int,
furDensity int,
birthdate datetime,
qualityType varchar(20));


CREATE TABLE mfCage
(cageNo int,
colNr int,
CageType varchar(20)
PRIMERY KEY (cageNo)
FOREIGN KEY (colNr) references mfColumn(minkID));


CREATE TABLE mfColumn
(colNo int,
numOfCol int,
cageNr int,
primary key (colNo)
foreign key (cageNr) references mfHall(cageNo);


CREATE TABLE mfHall
(hallNo int,
numOfCages int,
columnNo int,
farmCvr int,
primary key (hallID),
foreign key (farmCvr) references mfFarm(cvrno));
